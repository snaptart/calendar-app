Frontend folder structure for PHP modular architecture:

frontend/
├── components/           # Reusable UI components
│   ├── calendar/
│   │   ├── Calendar.php
│   │   └── EventModal.php
│   ├── tables/
│   │   ├── DataTable.php
│   │   └── UserTable.php
│   ├── forms/
│   │   ├── BaseForm.php
│   │   ├── LoginForm.php
│   │   └── ImportForm.php
│   ├── navigation/
│   │   ├── Header.php
│   │   ├── Navigation.php
│   │   └── Breadcrumb.php
│   └── ui/
│       ├── Modal.php
│       ├── StatusIndicator.php
│       └── FileUpload.php
├── layouts/              # Page layout templates
│   ├── BaseLayout.php
│   ├── AuthLayout.php
│   └── AppLayout.php
├── pages/               # Page controllers
│   ├── calendar.php
│   ├── events.php
│   ├── users.php
│   ├── import.php
│   └── login.php
├── services/            # API interface layer
│   ├── ApiClient.php
│   ├── AuthService.php
│   └── ConfigService.php
├── assets/              # Static assets (existing CSS/JS)
│   ├── css/
│   ├── js/
│   └── vendor/
└── config/              # Frontend configuration
    ├── app.php
    ├── components.php
    └── routes.php
